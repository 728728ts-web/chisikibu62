// ページが読み込まれた時に実行 window.onload = function() { // 保存されているカウントを取得 let count = localStorage.getItem('access_count'); if (count === null) { count = 1; // 初めての訪問 } else { count = parseInt(count) + 1; // 2回目以降 } // 数字を保存する localStorage.setItem('access_count', count); // 画面に表示する const display = document.getElementById('access-count'); if (display) { display.innerText = count; } };
